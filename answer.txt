var mapper = function() {
    emit (this.location.state , 1)
};

var reducer = function(key, value) {     return Array.sum(value)  };

db.contacts.mapReduce(
   mapper,
   reducer,
   {
     out: "result",
     query: {"gender": "male"},
        }
 );

db.result.find().sort({value: -1})

db.contacts.aggregate([ 
{ $match: { gender: "male" } },
{ $group: {  _id: { state: "$location.state"},  totalmale: { $sum: 1 }  }  },
{ $sort:  { totalmale: -1 } }  ] )
